# Divider

Extends [Material UI Divider](https://material-ui.com/components/dividers/).

Use a divider to separate different groups of content.

<Preview>
  <Story name="default">
    <Divider />
  </Story>
</Preview>

## Example use

<Preview>
  <Story name="card">
    <Box width={300}>
      <Card p={2}>
        <Flex mb={2} flexDirection="column">
          <Flex alignItems="baseline">
            <Box flexGrow={1}>
              <Heading.h4 gutterBottom={true}>Toothbrush</Heading.h4>
            </Box>
            <Box>
              <Heading.h6 gutterBottom>$4.50</Heading.h6>
            </Box>
          </Flex>
          <Paragraph color="textSecondary" light={true}>
            Pinstriped cornflower blue cotton blouse takes you on a walk to the
            park or just down the hall.
          </Paragraph>
        </Flex>
        <Divider variant="middle" />
        <Flex mt={2} mb={2} flexDirection="column">
          <Text gutterBottom={true}>Select type</Text>
          <Flex>
            <Chip clickable={false} m={1} color="primary" label="Soft" />
            <Chip clickable={true} m={1} label="Medium" />
            <Chip clickable={true} m={1} label="Hard" />
          </Flex>
        </Flex>
        <Box>
          <Button variant="text" color="primary">
            Add to cart
          </Button>
        </Box>
      </Card>
    </Box>
  </Story>
</Preview>
